57  	/**
58  	 * The paintComponent method from JComponent is here specialized to load all the images as specified on the map.
59  	 * By using a loop, we browse the table in order to check the value in each box
60  	 * A "switch case" then modify the imageName attribute dependently on the value found
61  	 * to make the image match the type of object (1:Wall, 2:Hero, ...)
62  	 * The function g.drawimage then draw the matching image on the JPanel.
63  	 */
64  	public void paintComponent(Graphics g) {
65  		try {
66  			
67  			//For a background image
68  			//Image backgroundImg = ImageIO.read(new File("backgroundImg.png"));
69  		    //g.drawImage(backgroundImg, 0, 0, this.getWidth(), this.getHeight(), this);

71  			/**
72  			 * The two loops "while()" are here used to browse all the table.
73  			 */
74  			
75      			X = 0;
76      			Y = 0;
77      		
78      		while(Y != 20) {
79      		
80      			while(X != 20) {
81      				
82      				switch (MAP[Y][X]) {
83      				
84      				/**
85      				 * The "switch case" is here used to modify the imagePath attribute depending of the type of object.
86      				 */
87      					case 1 :
88      						imagePath = "Wall.png";
89      					break;
90      					
91      					case 2 :
92      						imagePath = "Hero.png";
93      					break;
94      					
95      					case 3 :
96      						imagePath = "Boulder.png";
97      					break;
98      					
99      					case 4 :
100     						imagePath = "Diamond.png";
101     					break;
102     					
103     					case 5 :
104     						imagePath = "Dirt.png";
105     					break;
106     					
107     					case 6 :
108     						imagePath = "Enemy1.png";
109     					break;
110     					
111     					case 7 :
112     						imagePath = "EmptyBlock.png";
113     					break;
114     					
115     					case 8 :
116     						imagePath = "Enemy2.png";
117     					break;
118     					
119     					default :
120     						imagePath = "Error.png";
121     					break;
122     					
123     				}
124     				
125     				/**
126     				 * The two following methods are used to load the element depending of its path
127     				 * and draw it on the panel depending of its coordinates in the table. 
128     				 */
129     				Image img = ImageIO.read(new File(imagePath));
130     				g.drawImage(img, X*16, Y*16, this);
131 
132     				X++;
133     				
134     			}
135 
136     			Y++;
137     			X = 0;
138     		}
139 		} catch (IOException e){e.printStackTrace();}
140 	}
